; poisoned pixel        4390755
; healthy pixel         16751690
set %pets 3
set %healtpoint 60
set %emergency 48
;set %invis #true
set %invis #false
gosub init


while #true
{
  if %invis
    gosub invis
  for %i 1 %pets
  {
    gosub ispoisoned %PetID_ , %i
    if ( #result = #true )  && ( #CONTHP <= %emergency )
    {
      set #ltargetkind 2
      event macro 15 24 ; Cure Poison
      set #LTARGETID %PetID_ , %i
      target 5s
      event macro 22 0 ;target pet
      event SysMessage Curing #LTARGETID
      wait 5
    }
    if ( #result = #false )  && ( #CONTHP <= %healtpoint )
    {
      set #ltargetkind 1
      event macro 15 28 ; greater heal
      set #LTARGETID %PetID_ , %i
      target 5s
      event macro 22 0 ;target pet
      event SysMessage Healing #LTARGETID
    }
  }
}


sub ispoisoned

  event statbar %1
  wait 5
  set %contposx 38 + #CONTPOSX
  set %contposy 43 + #CONTPOSY
  savePix  %contposx %contposy 1
  wait 5
  if #PIXCOL = 4390755
  {
     return #true
  }

  if #PIXCOL = 16751690
  {
     return #false
  }

return error

sub init
;===========================
for %i 1 %pets
{
  Display ok Drag Pet %i healthbar where you want it - and target it.
  Set #TARGCURS 1
  While #TARGCURS = 1
  {
   Wait 1s
  }

  Set %PetID_ , %i #LTARGETID
  wait 5
  event statbar %PetID_ , %i
  wait 5
  set %PetID_ , %i , x #contposx
  set %PetID_ , %i , y #contposy
}
return

halt

sub invis
if H notin #CHARSTATUS
{
  event macro 15 43
  target 5s
  event macro 23
}
return
